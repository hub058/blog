# 5、编写内容

<configuration>
	<!-- 引入数据库连接的属性文件 -->
	<properties resource="database.properties"></properties>
	
	<!-- 配置数据库连接信息 -->
	<environments default="dev">
		<environment id="dev">
			<!-- JDBC自带的事务管理 -->
			<transactionManager type="JDBC"></transactionManager>
			<!-- 配置数据源：POOLED带连接池的数据源 -->
			<dataSource type="POOLED">
				<property name="driver" value="${driver}"/>
				<property name="url" value="${url}"/>
				<property name="username" value="${user}"/>
				<property name="password" value="${password}"/>
			</dataSource>
		</environment>
	</environments>

</configuration>

public class UserTest {
	
	Logger logger = Logger.getLogger(UserTest.class);
	
	@Test
	public void test() {
		try {
			//1.读取全局配置文件
			//ctrl+1(小键盘的数字):智能补全的快捷键
			Reader reader = Resources.getResourceAsReader("mybatis-config.xml");
			
			//2.创建SqlSessionFactory
			SqlSessionFactory factory = new SqlSessionFactoryBuilder().build(reader);
			
			//3.从工厂中获取会话对象SqlSession
			SqlSession session = factory.openSession();
			
			//4.使用session操作数据库
			Object count = session.selectOne("cn.smbms.dao.UserMapper.count");
			logger.info("用户总数是："+count);
			
		} catch (Exception e) {
			e.printStackTrace();
		}
	}

}

# 10、总结

MyBatis框架的优缺点：

优点：简单、代码量少、Sql语句和Java文件分离，解耦方便管理和维护

确定：编写SQL语句工作量大，需要熟悉SQL语句，数据库移植性差。

MyBatis专注于SQL本身，是一个足够灵活的DAO层解决方案，

适用于性能要求较高或者需求多变的互联网项目

# 11、MybatisUtil工具类

package cn.smbms.util;

import java.io.Reader;
import org.apache.ibatis.io.Resources;
import org.apache.ibatis.session.SqlSession;
import org.apache.ibatis.session.SqlSessionFactory;
import org.apache.ibatis.session.SqlSessionFactoryBuilder;

public class MybatisUtil {
	private static SqlSessionFactory factory = null;
	private static ThreadLocal<SqlSession> sessionTL = new ThreadLocal<SqlSession>();
	
	static{
		try {
			Reader reader = Resources.getResourceAsReader("mybatis-config.xml");
			factory = new SqlSessionFactoryBuilder().build(reader);
		} catch (Exception e) {
			e.printStackTrace();
		}
	}
	
	//打开Session的方法
	public static SqlSession currentSession(){
		SqlSession session = sessionTL.get();
		if(session==null){
			session = factory.openSession();
			sessionTL.set(session);
		}
		return session;
	}
	
	//关闭session的方法
	public static void closeSession(){
		SqlSession session = sessionTL.get();
		if(session!=null){
			session.close();
			sessionTL.set(null);
		}
	}

}
