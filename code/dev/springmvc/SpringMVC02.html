# 1、单例设计模式

概念

程序运行期间，有且只有一个实例，成为单例

# 1、实现原理

  • 一个类只有一个实例——最基本的

    • 只提供私有构造器

  • 它必须自行创建这个实例

    • 定义了静态的该类私有对象

  • 它必须自行向整个系统提供这个实例

    • 提供一个静态的公有方法，返回创建或者获取本身的静态私有对象

# 2、饿汉模式

/**
 * 造一个太阳，单例
 * @author Administrator
 *
 */
public class CreateSun {
	
	//2.静态属性,类加载的时候创建一个实例
	private static CreateSun sun = new CreateSun();

	//1.构造方法私有化
	private CreateSun() {
	}
	
	//3.提供给外界,饿汉模式，线程安全使用同步锁解决
	public synchronized static CreateSun getInstance(){
       if(sun == null){
			sun = new CreateSun();
		}
		return sun;
	}
	
}

# 3、懒汉模式

/**
 * 造一个太阳，单例
 * @author Administrator
 *
 */
public class CreateSun2 {
	
	//2.静态属性,类加载的时候先不创建实例
	private static CreateSun2 sun;

	//1.构造方法私有化
	private CreateSun2() {
	}
	
	//3.提供给外界，调用的时候才通过内部类创建实例
	public synchronized static CreateSun2 getInstance(){
		sun = CreateSunHelper.INSTANCE;
		return sun;
	}
    
	//使用静态内部类，保证了单例的同时，又有了延迟加载特性
	public static class CreateSunHelper {
		private static final CreateSun2 INSTANCE = new CreateSun2();
	}
	
}

@ExceptionHandler 的源码如下

@Target(ElementType.METHOD)
@Retention(RetentionPolicy.RUNTIME)
@Documented
public @interface ExceptionHandler {

	/**
	 * Exceptions handled by the annotation method. If empty, will default
	 * to any exceptions listed in the method argument list.
	 */
	Class<? extends Throwable>[] value() default {};

}

@Override
protected ModelAndView doResolveException(HttpServletRequest request, HttpServletResponse response, Object handler, Exception ex) {

    // Expose ModelAndView for chosen error view.
    String viewName = determineViewName(ex, request);
    if (viewName != null) {
        // Apply HTTP status code for error views, if specified.
        // Only apply it if we're processing a top-level request.
        Integer statusCode = determineStatusCode(request, viewName);
        if (statusCode != null) {
            applyStatusCodeIfPossible(request, response, statusCode);
        }
        return getModelAndView(viewName, ex, request);
    }
    else {
        return null;
    }
}
