# 1、构造注入

顾名思义：通过带参数的构造方法注入

  • 示例一(service层注入Dao层)
//dao
public class UserDaoImpl implements UserDao {

	@Override
	public void add(User user) {
		System.out.println("添加"+user.getUsername()+"到数据库中。。。");
	}

}
//service
public class UserServiceImpl implements UserService {
	private UserDao userDao;

	//建议保留无参构造-手动添加
	public UserServiceImpl() {
	}
	//提供带参构造
	public UserServiceImpl(UserDao userDao) {
		this.userDao = userDao;
	}
	
	@Override
	public void addUser(User user) {
		System.out.println("业务层添加用户");
		userDao.add(user);
	}

}

# 8、注解实现AOP

  • @Aspect //表示增强类
  • @Pointcut(value="execution(* cn.smbms.service...(..))") //配置切入点
  • @Before("pointcut()") //前置增强，引用了切入点，通过切入点的方法进行引用
  • @AfterReturning(returning="result",pointcut="pointcut()") //后置增强
  • @AfterThrowing(pointcut="pointcut()",throwing="e") //异常抛出增强
  • @Around(value="pointcut()") //环绕增强
  • @After("pointcut()") //最终增强
@Component("userServiceLogger")
@Aspect //表示增强类
public class UserServiceLogger {
	
    private static Logger log = Logger.getLogger(UserServiceLogger.class);
    
    //配置切入点
    @Pointcut(value="execution(* cn.smbms.service..*.*(..))")
    public void pointcut(){}
    
    @Before("pointcut()")
    public void before(JoinPoint jp) {
        log.info("前置增强的方法执行...");
        log.info("调用"+jp.getTarget()+"的"+jp.getSignature().getName()+"，入参是"+Arrays.toString(jp.getArgs()));
    }
    
    @AfterReturning(returning="result",pointcut="pointcut()")
    public void afterReturning(JoinPoint jp,Object result) {
    	log.info("后置增强的方法执行...");
    	log.info("调用"+jp.getTarget()+"的"+jp.getSignature().getName()+"，入参是"+jp.getArgs());
    	log.info("目标方法的返回值是..."+result);
    }
    
    //异常抛出增强
    @AfterThrowing(pointcut="pointcut()",throwing="e")
    public void afterThrowing(Exception e){
    	System.out.println("代码抛出了异常");
    	e.printStackTrace();
    }
    
    //环绕增强
    @Around(value="pointcut()")
    public Object around(ProceedingJoinPoint jp){
    	log.info("环绕增强的前置执行...");
        log.info("调用"+jp.getTarget()+"的"+jp.getSignature().getName()+"，入参是"+Arrays.toString(jp.getArgs()));
        Object object = null;
		try {
			object = jp.proceed(); //让目标方法继续执行
		} catch (Throwable e) {
			e.printStackTrace();
		}
        log.info("环绕增强的后置执行...");
    	log.info("调用"+jp.getTarget()+"的"+jp.getSignature().getName()+"，入参是"+jp.getArgs());
    	log.info("目标方法的返回值是..."+object);
    	return "nihao ";
    }
    
    //最终增强
    @After("pointcut()")
    public void after(){
    	System.out.println("最终增强的代码");
    }
}

# 9、@Value注解

给属性注入普通字面量的值

第二种方式：@Value("直接值")

第二种方式：@Value("${属性文件中的key}")

@Component("myUser")
public class User {
	
	//通过 @Value 注入字面量的值
	@Value("陈世杰")
	private String username;
	@Value("18")
	private Integer age;
	
	public String getUsername() {
		return username;
	}
	public void setUsername(String username) {
		this.username = username;
	}
	public Integer getAge() {
		return age;
	}
	public void setAge(Integer age) {
		this.age = age;
	}
	
    //测试代码
	public static void main(String[] args) {
		//前提:本类已通过组件扫描注入容器中
		ApplicationContext ac = new ClassPathXmlApplicationContext("applicationContext.xml");
        
		User user = (User) ac.getBean("myUser");
		System.out.println(user.getUsername());
}
